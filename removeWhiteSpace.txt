@echo off
setlocal enabledelayedexpansion

set "filename=%~1"  REM 파라미터로 전달된 파일 이름을 변수에 저장합니다.

if "%filename%"=="" (
  echo 파일 이름을 지정해야 합니다.
  exit /b
)

if not exist "%filename%" (
  echo 파일이 존재하지 않습니다.
  exit /b
)

> output.txt (
  for /f "usebackq delims=" %%a in ("%filename%") do (
    set "line=%%a"
    set "line=!line: =!"
    echo !line!
  )
)
move /y output.txt "%filename%"

echo 공백이 제거된 파일이 생성되었습니다.
